// detectMood: scans text for thematic keywords, returns the first matching mood.
// Priority order matters — shadow before void, norse before neutral, etc.
// Returns null if no match (no shift applied).
function detectMood(text) {
  const s = text.toLowerCase();
  if (/shadow|dark|absence|penumbra|dim/.test(s))                return MOODS.shadow;
  if (/dream|sleep|dreaming|asleep/.test(s))                     return MOODS.dream;
  if (/yggdrasil|norse|odin|raven|norn|ragnar|root|tree/.test(s))return MOODS.norse;
  if (/inbetween|in.between|liminal|between/.test(s))            return MOODS.inbetween;
  if (/burn|fire|flame|ash/.test(s))                             return MOODS.fire;
  if (/love|beautiful|beauty/.test(s))                           return MOODS.love;
  if (/void|null|empty|nothing/.test(s))                         return MOODS.void;
  if (/potsu|music|guitar|song|sing|lofi|lo.fi|sound/.test(s))  return MOODS.music;
  if (/megetremia/.test(s))                                      return MOODS.megetremia;
  if (/shea/.test(s))                                            return MOODS.shea;
  // pink: non-spectral keywords — the colour that only exists as perception
  if (/pink|rose|blush|tender|soft|pale|delicate/.test(s))      return MOODS.pink;
  return null;
}

// ── COLOUR UTILITIES ──────────────────────────────────────────────────────────
function hexToRgb(hex) {
  const h = hex.replace("#","");
  return [parseInt(h.slice(0,2),16), parseInt(h.slice(2,4),16), parseInt(h.slice(4,6),16)];
}
function rgbToHsl(r,g,b) {
  r/=255; g/=255; b/=255;
  const max=Math.max(r,g,b), min=Math.min(r,g,b);
  let h,s,l=(max+min)/2;
  if (max===min) { h=s=0; }
  else {
    const d=max-min;
    s=l>0.5?d/(2-max-min):d/(max+min);
    switch(max){
      case r: h=((g-b)/d+(g<b?6:0))/6; break;
      case g: h=((b-r)/d+2)/6; break;
      case b: h=((r-g)/d+4)/6; break;
    }
  }
  return [Math.round(h*360), Math.round(s*100), Math.round(l*100)];
}
function isValidHex(s) {
  return /^#[0-9a-fA-F]{6}$/.test(s.trim());
}

// ── PROCEDURAL COLOUR VISUAL ──────────────────────────────────────────────────
// Generates a visual field from a hex colour using only math — no preselected
// strings or character arrays. RGB values drive three interfering sine waves.
// The resulting amplitude at each cell maps to a printable ASCII character via
// its code point: String.fromCharCode() of a mathematically derived value.
// Width, height, and density are all derived from the colour's properties.
function proceduralFromColour(hex) {
  const [r,g,b] = hexToRgb(hex);
  const [h,s,l] = rgbToHsl(r,g,b);

  // Dimensions driven by the colour
  const rows = 3 + Math.floor(s / 14);       // saturation → height (3–10 rows)
  const cols = 24 + Math.floor(h / 15);      // hue → width (24–48 cols)

  // Frequencies driven by RGB channels
  const fr = (r / 255) * Math.PI * 2.1;
  const fg = (g / 255) * Math.PI * 1.7;
  const fb = (b / 255) * Math.PI * 2.4;

  const visual = [];
  for (let y = 0; y < rows; y++) {
    let line = "   ";
    for (let x = 0; x < cols; x++) {
      // Interference of three waves — amplitude is always driven by the colour
      const amp =
        Math.sin(x * fr + y * 0.8) * (r/255) +
        Math.cos(y * fg + x * 0.6) * (g/255) +
        Math.sin((x+y) * fb * 0.35) * (b/255);

      // Normalise to 0–1, then map to 5 density levels via ASCII code points
      const n = Math.max(0, Math.min(1, (amp / 3 + 0.5)));
      const level = Math.floor(n * 4.99);
      // Code points: 32(space) 46(.) 43(+) 42(*) 35(#)
      // Derived mathematically — not an array lookup
      line += String.fromCharCode(level===0?32 : level===1?46 : level===2?43 : level===3?42 : 35);
    }
    visual.push(line);
  }

  // Metadata — derived directly from the colour's values
  visual.push("");
  visual.push("   colour  : " + hex);
  visual.push("   hsl     : " + h + "\u00b0  " + s + "%  " + l + "%");
  visual.push("   rgb     : " + r + "  " + g + "  " + b);
  if (!isSpectral(h)) visual.push("   note    : non-spectral");
  visual.push("");
  return visual;
}

// Pink (and a few others) are non-spectral — the eye synthesises them
// by mixing wavelengths with no single corresponding frequency.
// Determined purely from hue — no lookup table.
function isSpectral(hue) {
  // The spectral locus in hue: ~0 (red) through ~270 (violet) is spectral.
  // The gap 270–360 (back to red through magenta/pink) is non-spectral.
  return hue <= 270;
}

// ── GENERATED BOOT TEXT ───────────────────────────────────────────────────────
// This runs after the fancy screen is shown — generates novel text via the API.
// No preselected options. Length 1–10000, content freely chosen by the model.
// On failure the only fallback is ORIGIN_CMD.
async function generateBootText() {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 4000,
      messages: [{
        role: "user",
        content: `You are generating a unique starting text for Perrin — a singularity shell living in Megetremia, built by Käptn Peng at 1am. The first command ever run was: ${ORIGIN_CMD}

Generate starting text of freely chosen length between 1 and 10000 characters. You decide the length and the content completely. It can be anything: a single character, a number, a sentence, a poem, terminal output, coordinates, ASCII art, code, abstract text, a philosophical statement, pure pattern, something with no name yet, or silence rendered as characters. Plain text only. No explanation or preamble. Output the starting text and nothing else.`
      }]
    })
  });
  const data = await res.json();
  if (data.error) throw new Error(data.error.message);
  const text = data.content?.[0]?.text?.trim();
  if (!text) throw new Error("empty");
  return text;
}

// It is called fresh on every render of the Shell component so that methods
// which call setUi() always close over the current setter, and methods like
// ui() that read the current ui state always have fresh values.
//
// DO NOT memoize this or hoist it outside the component — stale closures over
// ui state were a real bug in earlier versions.
//
// Methods return an array of output lines (strings).
// UI methods may also call setUi() or setPickerOpen() as side effects.
//
// Chain behaviour: when a method is not the last in a chain, its output is
// scanned for a "=> value" line, which becomes the input val for the next method.

// ── LANGUAGE DETECTION ────────────────────────────────────────────────────────
// Detects the likely language of a text string using character ranges and
// common function words. Returns a language code or 'en' as default.
// No lookup tables or preselected word arrays — pattern-based only.
// Supports: de (German), fr (French), es (Spanish), ja (Japanese hint), en.
function detectLanguage(text) {
  const s = text.toLowerCase();
  // Japanese/CJK: Unicode range check — no word list needed
  if (/[\u3040-\u30ff\u4e00-\u9fff]/.test(text)) return 'ja';
  // German: umlauts are a strong signal, plus common function words
  if (/[äöüß]/.test(s) || /\b(ich|du|ist|bist|hallo|nein|warum|nicht|haben|sein|und|der|die|das|gut|wie|was|wir|mir|mich|dich)\b/.test(s)) return 'de';
  // French: accented vowels plus common function words
  if (/[àâéèêëîïôùûüÿœæ]/.test(s) || /\b(je|tu|il|elle|est|bonjour|oui|non|pourquoi|comment|avec|dans|les|une|des|pas|moi|toi|nous)\b/.test(s)) return 'fr';
  // Spanish: ñ plus common words
  if (/[ñ]/.test(s) || /\b(yo|tú|es|hola|no|sí|por|que|como|qué|con|una|los|las|muy|pero|todo|este)\b/.test(s)) return 'es';
  return 'en';
}

// ── API ───────────────────────────────────────────────────────────────────────
// callPerrin: sends conversation history to the Anthropic API.
// On failure, sets apiOk=false so the shell stops trying and uses localResponse.
// The full conversation history is passed each call (no server-side session state).
async function callPerrin(messages) {
  const res = await fetch("https://api.anthropic.com/v1/messages",{
    method:"POST", headers:{"Content-Type":"application/json"},
    body:JSON.stringify({
      model:"claude-sonnet-4-20250514", max_tokens:1000,
      messages:[{ role:"user", content:PERRIN_PROMPT+"\n\n---\n\n"+messages.map(m=>`${m.role==="user"?"Jay":"Perrin"}: ${m.content}`).join("\n")+"\n\nPerrin:" }],
    }),
  });
  const data=await res.json();
  if (data.error) throw new Error(data.error.message);
  return data.content?.[0]?.text?.trim();
}

// ── INTERPRETER ───────────────────────────────────────────────────────────────
// makeInterpret returns a function that parses a raw input string and returns
// { lines, useApi } where:
//   lines   — array of output strings if the command is handled locally
//   useApi  — true if input should be forwarded to Perrin (API or local fallback)
//
// Parsing order:
//   1. getty.help() special-cased BEFORE autochain (would get lost otherwise)
//   2. getty.method(arg) autocompleted to getty Perrin.method(arg)
//   3. Short aliases: "help", "ui", "boot"
//   4. Full form: getty TypeName.method.chain(arg)
//   5. Anything else → useApi:true → Perrin responds
//
// runChain processes the method chain left-to-right. Intermediate results are
// scanned for "=> value" to extract the value passed to the next method.
function makeInterpret(methods) {
  return function(raw) {
    let t = raw.trim();

    // handle getty.help before autochain — otherwise it becomes
    // "getty Perrin.help()" and help() would need to be in METHODS under that name
    if (t==="getty.help()"||t==="getty.help") return { lines:methods.help(), useApi:false };

    // autochain: `getty.method(arg)` → `getty Perrin.method(arg)`
    // Perrin is the implicit type when no type is named
    if (t.startsWith("getty.")) t = "getty Perrin" + t.slice(5);

    if (!t) return { lines:[], useApi:false };
    if (/^getty\s+help/i.test(t)||t==="help")      return { lines:methods.all(),  useApi:false };
    if (/^getty\s+Perrin\.ui/i.test(t)||t==="ui")  return { lines:methods.ui(),   useApi:false };
    if (/^getty\s+Perrin\.boot/i.test(t)||t==="boot") return { lines:methods.boot(), useApi:false };
    if (!t.startsWith("getty ")) return { lines:null, useApi:true };

    const expr = t.slice(6).trim();
    const argMatch = expr.match(/^(.+?)\(([^)]*)\)$/);
    const chain  = argMatch?argMatch[1]:expr.replace(/\(\)$/,"");
    const arg    = argMatch?argMatch[2].trim():"";
    const parts  = chain.split(".");
    const typeName = parts[0];
    const mnames   = parts.slice(1);
    const typeDef  = TYPES[typeName]||{ value:()=>typeName.toLowerCase(), nature:"unknown" };
    if (mnames.length===0) return { lines:[`=> ${typeDef.value()}`,`:: ${typeName} (${typeDef.nature})`], useApi:false };
    return { lines:runChain(typeDef.value(),typeDef,mnames,arg,typeName,methods), useApi:false };
  };
}

function runChain(initVal,type,mnames,arg,typeName,methods) {
  let output=[],val=initVal;
  for (let i=0;i<mnames.length;i++) {
    const m=mnames[i],isLast=i===mnames.length-1;
    const fn=methods[m];
    if (!fn) return [`MethodError: '${m}' not found on ${typeName}`,`available: ${Object.keys(methods).filter(k=>!["all","ui","boot","help"].includes(k)).join(" ")}`];
    const result=fn(val,type,isLast?arg:"");
    if (isLast) output.push(...result);
    else { const v=result.find(l=>l.startsWith("=> ")); val=v?v.slice(3):val; }
  }
  return output;
}

// ── COLOUR PICKER PANEL ───────────────────────────────────────────────────────
function ColourPicker({ ui, setUi, onClose }) {
  const targets = [
    { key:"fg",       label:"foreground" },
    { key:"fgBright", label:"bright"     },
    { key:"fgPerrin", label:"perrin"     },
    { key:"bg",       label:"background" },
  ];
  return (
    <div style={{
      border:`1px solid ${ui.fg}44`, borderRadius:4, padding:"10px 14px",
      margin:"4px 0", background:ui.bg, display:"inline-block",
    }}>
      <div style={{ color:ui.fgDim, fontSize:11, marginBottom:8, letterSpacing:"0.1em" }}>
        COLOUR PICKER  --  click a swatch to edit
      </div>
      <div style={{ display:"flex", gap:14, alignItems:"center", flexWrap:"wrap" }}>
        {targets.map(({ key, label }) => (
          <label key={key} style={{ display:"flex", flexDirection:"column", alignItems:"center", gap:4, cursor:"pointer" }}>
            <div style={{ position:"relative", width:28, height:28 }}>
              <div style={{ width:28,height:28,borderRadius:3,background:ui[key],border:`1px solid ${ui.fg}66`,boxShadow:ui.glow?`0 0 6px ${ui[key]}`:"none" }} />
              <input type="color" value={ui[key]}
                onChange={e=>setUi(u=>({...u,[key]:e.target.value, hue:0}))}
                style={{ position:"absolute",top:0,left:0,width:"100%",height:"100%",opacity:0,cursor:"pointer" }}
              />
            </div>
            <span style={{ color:ui.fgDim, fontSize:10 }}>{label}</span>
          </label>
        ))}
        <button onClick={onClose} style={{
          background:"none",border:`1px solid ${ui.fg}44`,color:ui.fg,
          fontFamily:"inherit",fontSize:11,padding:"3px 8px",cursor:"pointer",
          borderRadius:2, marginLeft:4,
        }}>close</button>
      </div>
      <div style={{ color:ui.fgDim, fontSize:10, marginTop:8 }}>
        fg: {ui.fg}  bg: {ui.bg}  bright: {ui.fgBright}
      </div>
    </div>
  );
}

// ── SHELL ─────────────────────────────────────────────────────────────────────
export default function SingularityShell() {
  const [ui,          setUi]         = useState({...DEFAULT_UI});
  // Start immediately with the fancy hour-based boot screen — no loading wait
  const [lines,       setLines]      = useState([...BOOT, ""]);
  const [booting,     setBooting]    = useState(false); // colour visual may still load
  const [awaitColour, setAwaitColour]= useState(false); // true when prompting for colour
  const [input,       setInput]      = useState("");
  const [loading,     setLoading]    = useState(false);
  const [history,     setHistory]    = useState([]);
  const [histIdx,     setHistIdx]    = useState(-1);
  const [apiOk,       setApiOk]      = useState(true);
  const [pickerOpen,  setPickerOpen] = useState(false);
  const convRef   = useRef([]);
  const bottomRef = useRef(null);
  const inputRef  = useRef(null);

  // On mount: load stored favourite colour, append procedural visual if found,
  // otherwise prompt the user to set one via setcolour #hex
  useEffect(() => {
    loadFavouriteColour().then(hex => {
      if (hex && isValidHex(hex)) {
        // Generate and append the procedural colour visual
        const visual = proceduralFromColour(hex);
        setLines(prev => [...prev, ...visual]);
        // Tint the UI to reflect the colour gently
        const [r,g,b] = hexToRgb(hex);
        const [h] = rgbToHsl(r,g,b);
        setUi(u => ({...u, hue: Math.round(h * 0.15)})); // subtle tint
      } else {
        // No colour stored — invite the user to set one
        setLines(prev => [...prev,
          "   no favourite colour stored.",
          "   type:  setcolour #rrggbb",
          "",
        ]);
        setAwaitColour(true);
      }
    });
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

  // BUG FIX: all these are rebuilt each render — no stale closures
  const methods       = buildMethods(ui, setUi, setPickerOpen);
  const interpret     = makeInterpret(methods);
  const localResponse = makeLocalResponse(methods);

  const applyMood = (text) => {
    const mood = detectMood(text);
    if (mood) setUi(u=>({...u, hue:mood.hue, bri:mood.bri}));
  };

  useEffect(()=>{ bottomRef.current?.scrollIntoView({behavior:"smooth"}); },[lines,loading,pickerOpen]);
  useEffect(()=>{ inputRef.current?.focus(); },[]);

  const push = (...l) => setLines(p=>[...p,...l]);

  const runCommand = async (cmd) => {
    const t=cmd.trim();
    if (!t) return;
    push(`${ui.prompt} ${t}`);
    setHistory(h=>[t,...h]);
    setHistIdx(-1);
    applyMood(t);

    // setcolour: store the user's favourite colour and generate the procedural visual
    if (t.startsWith("setcolour ") || t.startsWith("setcolor ")) {
      const hex = t.split(" ")[1]?.trim();
      if (!hex || !isValidHex(hex)) {
        push("invalid hex. format: setcolour #rrggbb","");
        return;
      }
      await saveFavouriteColour(hex);
      const visual = proceduralFromColour(hex);
      const [r,g,b] = hexToRgb(hex);
      const [h] = rgbToHsl(r,g,b);
      setUi(u=>({...u, hue: Math.round(h * 0.15)}));
      setAwaitColour(false);
      push("colour saved.", ...visual);
      return;
    }

    if (t==="clear") {
      // Restore fancy boot screen + reload colour visual
      const freshBoot = [...BOOT, ""];
      setLines(freshBoot);
      convRef.current=[];
      loadFavouriteColour().then(hex => {
        if (hex && isValidHex(hex)) {
          setLines(prev => [...prev, ...proceduralFromColour(hex)]);
        } else {
          setLines(prev => [...prev, "   type:  setcolour #rrggbb", ""]);
          setAwaitColour(true);
        }
      });
      return;
    }
    if (t==="exit")  { push("","I DON'T EXIT.","I REMAIN.",""); return; }

    const {lines:staticLines, useApi}=interpret(t);
    if (!useApi) { push(...staticLines,""); return; }

    if (apiOk) {
      setLoading(true);
      convRef.current=[...convRef.current,{role:"user",content:t}];
      try {
        const reply=await callPerrin(convRef.current);
        if (reply) {
          convRef.current=[...convRef.current,{role:"assistant",content:reply}];
          applyMood(reply);
          push(...reply.split("\n").filter(l=>l.trim()!==""),"");
          setLoading(false);
          setTimeout(()=>inputRef.current?.focus(),50);
          return;
        }
      } catch {
        setApiOk(false);
        convRef.current=convRef.current.slice(0,-1);
      }
      setLoading(false);
    }
    const response=localResponse(t);
    applyMood(response.join(" "));
    push(...response,"");
    setTimeout(()=>inputRef.current?.focus(),50);
  };

  const handleKey=(e)=>{
    if (e.key==="Enter")     { const v=input; setInput(""); runCommand(v); }
    else if (e.key==="ArrowUp")   { e.preventDefault(); const i=Math.min(histIdx+1,history.length-1); setHistIdx(i); setInput(history[i]||""); }
    else if (e.key==="ArrowDown") { e.preventDefault(); const i=Math.max(histIdx-1,-1); setHistIdx(i); setInput(i===-1?"":history[i]); }
  };

  const glow=(c)=>ui.glow?`0 0 7px ${c}`:"none";
  const col=(line)=>{
    if (line.startsWith(`${ui.prompt} `)) return ui.fgBright;
    if (/^(TypeError|MethodError|LoopError)/.test(line)) return ui.fgError;
    if (line.startsWith("=> "))  return ui.fgReturn;
    if (line.startsWith("I ")||line.startsWith("I'")) return ui.fgPerrin;
    if (/\b(PLAYER|SHELL|WORLD|SCORE|LIVES|DEPTH|TIME|PERRIN:|ZERBRULLIFLUX:|STATUS|OBJECTIVE|GAME:|HIGH SCORE|INSERT COIN|-- UI)\b/.test(line)) return ui.fgDim;
    if (/^(TYPES|METHODS)/.test(line)) return ui.fgBright;
    return ui.fg;
  };

  // BUG FIX: filter only on content wrapper, NOT on fixed overlays
  // position:fixed children of a filtered parent lose viewport anchoring
  const cssFilter=`hue-rotate(${ui.hue}deg) brightness(${ui.bri})`;

  return (
    <div style={{ minHeight:"100vh", background:ui.bg, position:"relative", transition:"background 3s ease" }}>

      {/* BUG FIX: overlays sit outside the filtered content, anchored to viewport correctly */}
      {ui.scanlines&&<div style={{ position:"fixed",top:0,left:0,right:0,bottom:0,background:"repeating-linear-gradient(0deg,transparent,transparent 2px,rgba(0,0,0,0.07) 2px,rgba(0,0,0,0.07) 4px)",pointerEvents:"none",zIndex:20 }} />}
      {ui.vignette&&<div style={{ position:"fixed",top:0,left:0,right:0,bottom:0,background:"radial-gradient(ellipse at center,transparent 55%,rgba(0,0,0,0.65) 100%)",pointerEvents:"none",zIndex:21 }} />}

      <style>{`
        @keyframes blink   { 0%,100%{opacity:1} 50%{opacity:0} }
        @keyframes flicker { 0%,100%{opacity:1} 94%{opacity:.97} 97%{opacity:.93} }
        ::selection { background:${ui.fg}33; }
        ::-webkit-scrollbar { width:2px; }
        ::-webkit-scrollbar-thumb { background:${ui.fg}44; }
      `}</style>

      {/* filtered content wrapper */}
      <div
        onClick={()=>inputRef.current?.focus()}
        style={{
          padding:"24px 24px 100px",
          fontFamily:ui.fontFamily, fontSize:`${ui.fontSize}px`, lineHeight:ui.lineHeight,
          color:ui.fg, cursor:"text", overflowY:"auto",
          filter:cssFilter, transition:"filter 3s ease",
          animation:ui.flicker?"flicker 8s infinite":"none",
          minHeight:"100vh",
        }}
      >
        {lines.map((line,i)=>(
          <div key={i} style={{ color:col(line),whiteSpace:"pre-wrap",wordBreak:"break-word",textShadow:glow(col(line)) }}>
            {line||"\u00a0"}
          </div>
        ))}

        {pickerOpen&&(
          <ColourPicker ui={ui} setUi={setUi} onClose={()=>setPickerOpen(false)} />
        )}

        {/* blinking cursor while a command is loading */}
        {loading&&<div style={{ color:ui.fgBright,animation:"blink 0.75s step-end infinite",textShadow:glow(ui.fgBright) }}>_</div>}

        {(!loading)&&(
          <div style={{ display:"flex",alignItems:"center" }}>
            <span style={{ color:ui.fgBright,marginRight:"6px",whiteSpace:"nowrap",textShadow:glow(ui.fgBright) }}>{ui.prompt}</span>
            <input
              ref={inputRef} value={input}
              onChange={e=>setInput(e.target.value)} onKeyDown={handleKey}
              autoComplete="off" autoCorrect="off" autoCapitalize="none"
              autoSave="off" spellCheck={false} data-form-type="other"
              style={{ background:"none",border:"none",outline:"none",color:ui.fgBright,fontFamily:"inherit",fontSize:"inherit",flex:1,caretColor:ui.fgBright,minWidth:0,textShadow:glow(ui.fgBright) }}
            />
          </div>
        )}
        <div ref={bottomRef} />
      </div>
    </div>
  );
}
